        private void ValidareCampuriRealTime(object sender, TextChangedEventArgs e)
        {
            TextBox textBox = sender as TextBox;
            if (textBox == null) return;

            if (textBox == txtPret)
            {
                if (!string.IsNullOrEmpty(txtPret.Text) && !Regex.IsMatch(txtPret.Text, @"^\d+(\.\d{0,2})?$"))
                {
                    txtPret.BorderBrush = System.Windows.Media.Brushes.Red;
                    txtStatus.Text = "Pretul trebuie sa fie un numar valid";
                }
                else
                {
                    txtPret.BorderBrush = System.Windows.Media.Brushes.Gray;
                    txtStatus.Text = "Gata";
                }
            }
            else if (textBox == txtGarantie)
            {
                if (!string.IsNullOrEmpty(txtGarantie.Text) && !Regex.IsMatch(txtGarantie.Text, @"^\d+$"))
                {
                    txtGarantie.BorderBrush = System.Windows.Media.Brushes.Red;
                    txtStatus.Text = "Garantia trebuie sa fie un numar intreg";
                }
                else
                {
                    txtGarantie.BorderBrush = System.Windows.Media.Brushes.Gray;
                    txtStatus.Text = "Gata";
                }
            }
            else if (textBox == txtDenumire || textBox == txtCategorie || textBox == txtBrand)
            {
                if (textBox.Text.Length > 50)
                {
                    textBox.BorderBrush = System.Windows.Media.Brushes.Orange;
                    txtStatus.Text = "Textul este prea lung (max 50 caractere)";
                }
                else
                {
                    textBox.BorderBrush = System.Windows.Media.Brushes.Gray;
                    txtStatus.Text = "Gata";
                }
            }
        }

        private void btnAdaugaA_Click(object sender, RoutedEventArgs e)
        {
            if (ValidareDate())
            {
                if (VerificaDuplicat(listaA))
                {
                    MessageBox.Show("Exista deja un gadget cu aceasta denumire in Lista A!", "Duplicat", MessageBoxButton.OK, MessageBoxImage.Warning);
                    return;
                }

                listaA.Add(CreareGadget());
                ActualizeazaPaginareA();
                CurataCampuri();
                txtStatus.Text = "Gadget adaugat in Lista A";
            }
        }

        private void btnAdaugaB_Click(object sender, RoutedEventArgs e)

        private bool VerificaDuplicat(List<Gadget> lista)
        {
            return lista.Any(g => g.Denumire.Equals(txtDenumire.Text.Trim(), StringComparison.OrdinalIgnoreCase));
        }

        private bool ValidareDate()
        {
            if (string.IsNullOrWhiteSpace(txtDenumire.Text))
            {
                MessageBox.Show("Introduceti denumirea!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtDenumire.Focus();
                return false;
            }

            if (txtDenumire.Text.Trim().Length < 3)
            {
                MessageBox.Show("Denumirea trebuie sa aiba cel putin 3 caractere!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtDenumire.Focus();
                return false;
            }

            if (txtDenumire.Text.Trim().Length > 50)
            {
                MessageBox.Show("Denumirea nu poate depasi 50 de caractere!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtDenumire.Focus();
                return false;
            }

            if (string.IsNullOrWhiteSpace(txtCategorie.Text))
            {
                MessageBox.Show("Introduceti categoria!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtCategorie.Focus();
                return false;
            }

            if (txtCategorie.Text.Trim().Length < 3)
            {
                MessageBox.Show("Categoria trebuie sa aiba cel putin 3 caractere!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtCategorie.Focus();
                return false;
            }

            if (string.IsNullOrWhiteSpace(txtBrand.Text))
            {
                MessageBox.Show("Introduceti brandul!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtBrand.Focus();
                return false;
            }

            if (txtBrand.Text.Trim().Length < 2)
            {
                MessageBox.Show("Brandul trebuie sa aiba cel putin 2 caractere!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtBrand.Focus();
                return false;
            }

            decimal pret;
            if (!decimal.TryParse(txtPret.Text, out pret))
            {
                MessageBox.Show("Introduceti un pret valid!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtPret.Focus();
                return false;
            }

            if (pret < 0)
            {
                MessageBox.Show("Pretul nu poate fi negativ!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtPret.Focus();
                return false;
            }

            if (pret > 100000)
            {
                MessageBox.Show("Pretul este prea mare (max 100,000 lei)!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtPret.Focus();
                return false;
            }

            int garantie;
            if (!int.TryParse(txtGarantie.Text, out garantie))
            {
                MessageBox.Show("Introduceti o garantie valida!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtGarantie.Focus();
                return false;
            }

            if (garantie < 0)
            {
                MessageBox.Show("Garantia nu poate fi negativa!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtGarantie.Focus();
                return false;
            }

            if (garantie > 120)
            {
                MessageBox.Show("Garantia nu poate depasi 120 de luni!", "Validare", MessageBoxButton.OK, MessageBoxImage.Warning);
                txtGarantie.Focus();
                return false;
            }

            return true;
        }

        private void ActualizeazaPaginareA()
        {
            if (listaA == null)
            {
                listaA = new List<Gadget>();
            }
            
            int totalPagini = dimensiunePaginaA > 0 ? (int)Math.Ceiling((double)listaA.Count / dimensiunePaginaA) : 1;
            if (totalPagini == 0) totalPagini = 1;

            if (paginaCurentaA > totalPagini) paginaCurentaA = totalPagini;

            var listaPaginata = dimensiunePaginaA > 0
                ? listaA.Skip((paginaCurentaA - 1) * dimensiunePaginaA).Take(dimensiunePaginaA).ToList()
                : listaA.ToList();

            lstA.ItemsSource = null;
            lstA.ItemsSource = listaPaginata;

            txtPaginaA.Text = $"Pagina {paginaCurentaA} / {totalPagini} (Total: {listaA.Count})";

            btnPrevA.IsEnabled = paginaCurentaA > 1;
            btnNextA.IsEnabled = paginaCurentaA < totalPagini;
        }

        private void SorteazaListaA(string criteriu)
        {
            switch (criteriu)
            {
                case "DenumireAsc":
                    listaA = listaA.OrderBy(g => g.Denumire).ToList();
                    break;
                case "DenumireDesc":
                    listaA = listaA.OrderByDescending(g => g.Denumire).ToList();
                    break;
                case "PretAsc":
                    listaA = listaA.OrderBy(g => g.Pret).ToList();
                    break;
                case "PretDesc":
                    listaA = listaA.OrderByDescending(g => g.Pret).ToList();
                    break;
                case "GarantieAsc":
                    listaA = listaA.OrderBy(g => g.GarantieLuni).ToList();
                    break;
                case "GarantieDesc":
                    listaA = listaA.OrderByDescending(g => g.GarantieLuni).ToList();
                    break;
            }
            paginaCurentaA = 1;
            ActualizeazaPaginareA();
        }

        private void btnGrupare_Click(object sender, RoutedEventArgs e)
        {
            if (cmbFiltruGrupare.SelectedItem == null)
            {
                MessageBox.Show("Selectati un criteriu de grupare!", "Info", MessageBoxButton.OK, MessageBoxImage.Information);
                return;
            }

            string criteriu = (cmbFiltruGrupare.SelectedItem as ComboBoxItem).Tag.ToString();
            string listaSelectata = (cmbListaGrupare.SelectedItem as ComboBoxItem).Tag.ToString();

            List<Gadget> listaDeGrupat = new List<Gadget>();

            if (listaSelectata == "A")
            {
                listaDeGrupat = listaA.ToList();
            }
            else if (listaSelectata == "B")
            {
                listaDeGrupat = listaB.ToList();
            }
            else if (listaSelectata == "R")
            {
                var reuniune = listaA.Union(listaB).ToList();
                listaDeGrupat = reuniune;
            }

            if (listaDeGrupat.Count == 0)
            {
                MessageBox.Show("Nu exista date pentru grupare!", "Info", MessageBoxButton.OK, MessageBoxImage.Information);
                return;
            }

            List<GrupareInfo> grupuri = new List<GrupareInfo>();

            if (criteriu == "Brand")
            {
                var grupate = listaDeGrupat.GroupBy(g => g.Brand).OrderBy(gr => gr.Key);
                foreach (var grup in grupate)
                {
                    grupuri.Add(new GrupareInfo
                    {
                        Nume = $"{grup.Key}: {grup.Count()}",
                        Elemente = grup.Select(g => g.Denumire).ToList()
                    });
                }
            }
            else if (criteriu == "Categorie")
            {
                var grupate = listaDeGrupat.GroupBy(g => g.Categorie).OrderBy(gr => gr.Key);
                foreach (var grup in grupate)
                {
                    grupuri.Add(new GrupareInfo
                    {
                        Nume = $"{grup.Key}: {grup.Count()}",
                        Elemente = grup.Select(g => g.Denumire).ToList()
                    });
                }
            }
            else if (criteriu == "Pret")
            {
                var grupate = listaDeGrupat.GroupBy(g =>
                {
                    if (g.Pret < 500) return "0-500 lei";
                    else if (g.Pret < 1000) return "500-1000 lei";
                    else if (g.Pret < 2000) return "1000-2000 lei";
                    else return "2000+ lei";
                }).OrderBy(gr => gr.Key);

                foreach (var grup in grupate)
                {
                    grupuri.Add(new GrupareInfo
                    {
                        Nume = $"{grup.Key}: {grup.Count()}",
                        Elemente = grup.Select(g => $"{g.Denumire} ({g.Pret:F2} lei)").ToList()
                    });
                }
            }

            lstGrupare.ItemsSource = grupuri;
            txtTotalGrupare.Text = $"TOTAL: {listaDeGrupat.Count}";
            txtStatus.Text = $"Grupare realizata: {grupuri.Count} grupuri";
        }

        private void btnAgregate_Click(object sender, RoutedEventArgs e)
        {
            var reuniune = listaA.Union(listaB).ToList();

            if (reuniune.Count == 0)
            {
                MessageBox.Show("Nu exista date pentru a calcula agregate!", "Info", MessageBoxButton.OK, MessageBoxImage.Information);
                return;
            }

            int countA = listaA.Count;
            int countB = listaB.Count;
            int countReuniune = reuniune.Count;

            decimal pretMediu = reuniune.Average(g => g.Pret);
